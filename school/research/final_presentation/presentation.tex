        \begin{enumerate}
            \item \textbf{- Requires local operational ordering and bounded
                system size.}
            \item \textbf{- Implies stability.}
        \end{enumerate}

        Eager Collection
        \begin{enumerate}
            \item - Requires local operational ordering and bounded
                system size
            \item - Tracks and pushes stability.
